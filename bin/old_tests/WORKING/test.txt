#!/bin/bash

D_OUTPUT=($(grep D_e Si_EAM.tmp | awk '{print $2}'))
A_OUTPUT=($(grep a Si_EAM.tmp | awk '{print $2}'))
R_OUTPUT=($(grep r_e Si_EAM.tmp | awk '{print $2}'))

INDEX=1
MAX=${#D_OUTPUT[@]}
##echo "max: $MAX"

while [ $INDEX -lt $MAX ]; do
    
    echo -e "${D_OUTPUT[$INDEX]}\t${A_OUTPUT[$INDEX]}\t${R_OUTPUT[$INDEX]}\r"       >>output_1.txt
    echo -e "${D_OUTPUT[$INDEX+1]}\t${A_OUTPUT[$INDEX+1]}\t${R_OUTPUT[$INDEX+1]}\r" >>output_2.txt
    echo -e "${D_OUTPUT[$INDEX+2]}\t${A_OUTPUT[$INDEX+2]}\t${R_OUTPUT[$INDEX+2]}\r" >>output_3.txt

    let INDEX=INDEX+3
done
